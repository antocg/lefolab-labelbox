import csv

with open(r'projects\2024_bci\BCNM_SPECIES_BOTANISTS_LIST_2026-04-30.csv', encoding='utf-8-sig') as f:
    reader = csv.DictReader(f, delimiter=';')
    matched_genera = set()
    accepted_genera = set()
    current_genera = set()
    for row in reader:
        m = row.get('wcvp_matched_name', '').strip()
        a = row.get('wcvp_accepted_name', '').strip()
        c = row.get('current_binomial', '').strip()
        if m: matched_genera.add(m.split()[0])
        if a: accepted_genera.add(a.split()[0])
        if c: current_genera.add(c.split()[0])
